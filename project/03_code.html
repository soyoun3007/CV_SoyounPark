import cv2

input_video = "images/scratch_assay.mp4"
vidcap = cv2.VideoCapture(input_video)

def getFrame(sec): 
    vidcap.set(cv2.CAP_PROP_POS_MSEC, sec*1000) 
    hasFrames,image = vidcap.read() 
    if hasFrames: 
        cv2.imwrite(f"images/frame_{str(float(sec))}sec.jpg", image) # save frame as JPG file 
    return hasFrames 

frameRate = 9.5
for i in range(0, 5):
    sec = i*frameRate
    getFrame(sec)

In [6]:
fig, axes = plt.subplots(ncols=len(file_list), sharex=True, sharey=True, figsize=(16, 16))

time = 0
time_list = []
area_list = []

for file in file_list:
    img = io.imread(file, as_gray=True)
    img_shape = img.shape
    x_cut = 40
    img_cut = img_as_ubyte(img[0:img_shape[0], x_cut:img_shape[1]-x_cut]) # remove right/left sides of imgage
    entropy_img = entropy(img_cut, disk(3))
    thresh = threshold_otsu(entropy_img)
    binary = entropy_img <= thresh
    area_list.append(np.sum(binary == True)) # caculate the area
    time_list.append(time*6)
    
    # check binary images
    axes[time].imshow(binary)
    axes[time].set_title(f"{time_list[time]} hr")
    
    time += 1

import pandas as pd
import seaborn as sns
from scipy.stats import linregress

# dataframe with time and area
data = pd.DataFrame(time_list, area_list).reset_index()
data = data.rename(columns = {"index":"area", 0:"hour"})
data["log_area"] = np.log10(data["area"])

# make a figure
param = {"axes.spines.right": False, "axes.spines.top": False}
sns.set_theme(style="white", rc=param)
fig = sns.regplot(data=data, x="hour", y="log_area", 
                  scatter_kws={"edgecolors":"black", "s":50},
                 line_kws={"color":"lightcoral", "linestyle":"--"})

# labels
fig.set_xlabel("Time (hours)", fontsize=15, weight="bold")
fig.set_ylabel("log10(Gap Area)", fontsize=15, weight="bold")
fig.set_title("Cell Migration Assay (24 hours)", fontsize=20, weight="bold")

# linear regression 
result = linregress(data.hour, data.log_area)
text = f"y = {round(result.slope, 4)}x + {round(result.intercept, 4)} \n p = {round(result.pvalue, 2)} \n r = {round(result.rvalue, 2)}"
plt.figtext(0.7, 0.65, text, ha="center", fontsize=15)
